       psi~dunif(0,1)

       for(i in 1:(nind+nz)){
          z[i]~dbern(psi) # latent indicator variables from data augmentation
          x[i]~dunif(0,4) # distance is a random variable
          logp[i]<- -((x[i]*x[i])/theta2)
          p[i]<-exp(logp[i])
          mu[i]<-z[i]*p[i]
          y[i]~dbern(mu[i]) # observation model
       }
       N<-sum(z[1:(nind+nz)])
